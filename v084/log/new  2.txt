def proc_view_field_replyinput_sno_trn_dflt_for_tbl_set
	ordtblname = command_c[:sio_viewname].splt("_")[1].sub("inst","ord")
	ordtblid = eval("@" + ordtblname.chop + "_id")
	strsql = "select count(*) cnt from alloctbls  where destblname = '#{ordtblname}' and  desttblid = '#{ordtblid}'
            and srctblname = 'command_c[:sio_viewname].splt("_")[1]'
			group by destblname,destblid"
	next_sno = ActiveRecord::Base.connection.select_one(strsql)
	if  next_sno
		sno = eval(eval("@" + ordtblname.chop + "_sno") + format("%03d",(next_sno["cnt"]+ 1))
	else
		sno = eval(eval("@" + ordtblname.chop + "_sno") + "001"
	end
	return sno
end
